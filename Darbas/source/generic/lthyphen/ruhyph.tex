\def\MapInpEncToFntEnc#1#2{
  \bgroup
    \uccode 0=#1
    \uccode 1=#2
    \uppercase{
  \egroup
      \let ^^00=^^01    % this is \let <active char #1>=<letter #2>
    }
  \lccode #2=#2
  \catcode #1=13
}
